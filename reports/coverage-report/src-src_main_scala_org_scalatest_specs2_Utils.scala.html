40   /**
41    * Creates a nice suite name for the given specs2 specification instance.
42    *
43    * <p>
44    * The suite name is used to display on the <i>ScalaTest View</i>.
45    * </p>
46    *
47    * <p>
48    * The specification title is more appropriate than the class name for a
49    * <code>suiteName</code> because the user can specify a more readable name
50    * for a specification, so it is accessed as a first try.
51    * </p>
52    *
53    * @param spec either an acceptance specification or a mutable (unit) specification;
54    *         <i>cannot be null</i>
55    * @return the <i>title</i> of the specification if it was set by the user;
56    *         <b>or</b> a bit decorated version of the name of the given
57    *         <code>spec</code> (if applicable)
58    */
59   def suiteNameFor(spec: SpecificationStructure): String = {
60     require(spec != null)
61
62     if (hasTitle(spec)) {
63       return spec.identification.title
64     } else {
65       val baseName = spec.identification.name
66
67       if (spec.isInstanceOf[Specification]) {
68         return baseName + " Acceptance Specification"
69       } else if (spec.isInstanceOf[org.specs2.mutable.Specification]) {
70         return baseName + " Unit Specification"
71       } else {
72         // Nothing should fall into this category, but a default value is always good to have
73         return baseName + " Specification"
74       }
75     }
76   }
